Table Users {
  id integer [primary key]
  email varchar
  username varchar
  hashed_password varchar
  isChurchgoer boolean
  created_at timestamp
  updated_at timestamp
}

Table Churches {
  id integer [primary key]
  user_id integer
  name varchar
  latitude integer
  longitude integer
  address varchar
  city varchar
  state varchar
  description text
  phone varchar
  email varchar
  website varchar
  created_at timestamp
  updated_at timestamp
}

Table Reviews {
  id integer [primary key]
  user_id integer
  church_id integer
  body text
  rating integer
  created_at timestamp
  updated_at timestamp
}

Table UserPreferences {
  id integer [primary key]
  user_id integer
  preferred_size integer  // target average size
  preferred_age_group text
  preferred_ethnic_majority text
  preferred_language varchar
  preferred_denomination text
  wants_to_serve boolean
  created_at timestamp
  updated_at timestamp
}

Table ChurchAttributes {
  id integer [primary key]
  church_id integer
  size integer  // Number of regular attendees
  denomination text
  language varchar
  ethnic_majority text  // e.g., "Multicultural", "Predominantly Black", etc.
  age_group text  // e.g., "Youth-focused", "Family", "Seniors", etc.
  offers_service_opportunities boolean
  created_at timestamp
  updated_at timestamp
}

Table ChurchImages {
  id integer [primary key]
  church_id integer
  url varchar
  created_at timestamp
  updated_at timestamp
}

Table SavedChurches {
  id integer [primary key]
  user_id integer
  church_id integer
  created_at timestamp
}

Table ServiceOpportunities {
  id integer [primary key]
  church_id integer
  title varchar
  description text
  is_active boolean
  created_at timestamp
  updated_at timestamp
}
